## 替换日期并更换月日位置
```javascript
    var a = '2018-12-11，年后2019-11-11';
    var b = a.matchAll(/(\d{4})-(\d{2})-(\d{2})/g);
    while (!b.next().done) {
        const v = b.next().value[0];
        c.push(
            {
            c: v,
            y: v.split('-')[0] || b.next().value[1],
            m: v.split('-')[1] || b.next().value[2],
            d: v.split('-')[2] || b.next().value[3],
            }
        )
    }
    var d = a;
    c.forEach(v => {
        d = d.replace(v.c, `${v.y}/${v.d}/${v.m}`);
    });
```
