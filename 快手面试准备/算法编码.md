## 反转链表
```javascript
    const reverseList = function(head) {
        if (!head || !head.next) {
            return head;
        }

        let cur = head;
        let pre = null;
        let next = null;

        while(cur) {
            next = cur.next;
            cur.next = pre;
            pre = cur;
            cur = next;
        }

        return pre;
    }
```


## 全排列
```javascript
    
    const permute = function(nums) {
        const res = [], path = [];
        backtracking(nums, nums.length, {});
        return res;


        backtracking = function(nums, max, used) {
            if (path.length === max) {
                res.push(Array.from(path));
                return
            }

            // 逐个数字进行递归
            for(let i = 0; i < max; i++) {
                if (used[i]) continue;
                path.push(nums[i]);
                used[i] = true;
                backtracking(nums. max, used);
                // 回溯 匹配出所有可能 再往下走
                path.pop();
                used[i] = false;
            }
        }
    }

```

## 防抖 / 节流
```javascript

    function debounce(fn, delay) {
        let timer = null;

        return function() {
            const context = this;
            const args = arguments;
            clearTimeout(timer);
            timer = setTimeout(function() {
                fn.apply(context, args);
            }, delay);
        }
    }


    function throttle(func, delay) {
        let timer = null;
        return function() {
            let context = this;
            let args = arguments;
            if (!timer) {
                time = setTimeout(function() {
                    func.apply(context, args);
                    time = null;
                }, delay);
            }
        }
    }

```

## 发布订阅

```javascript

    const Observe = {
        consturctor() {
            // 缓存列表
            this.listener = {};
        }

        // 订阅
        on(eventName, fn) {
            if (!this.listener[eventName]) {
                this.listener[eventName] = [];
            }
            this.listener[eventName].push(fn)
        }

        // 发布
        emit(eventName, data) {
            const callbacks = this.listener[eventName];
            if (callbacks) {
                callbacks.forEach(func => {
                    func(data);
                });
            }
        }

        // 清除订阅
        off(eventName, fn) {
            const callbacks = this.listener[eventName];
            if (!callbacks) {
                return false;
            }
            if (!fn) {
                callbacks.length = 0;
            } else {
                let cb;
                for (let i = 0; i < callback.length; i++) {
                    cb = callbacks[i];
                    if (cb == fn || cb.fn == fn) {
                        callbacks.splice(i, 1);
                        break;
                    }
                }
            }
        }
    }

```

## 观察者模式
```javascript

    function Observer(obj, targetVariable, callback) {
        if (!obj.data) {
            obj.data = {}
        }
        Object.defineProperty(obj, targetVariable, {
            get() {
                return this.data[targetVariable]
            },
            set(val) {
                this.data[targetVariable] = val;
                // 目标主动通知观察者
                callback && callback(val)
            }
        });
        if (obj.data[targetVariable]) {
            callback && callback(obj.data[targetVariable]);
        }
    }

```

## call / apply / bind
```javascript


    // call
    Function.prototype.call = function(context) {
        if (typeof this !== 'function') {
            return 'error';
        }
        context = context || window;
        context.fn = this;
        const args = [...arguments].slice(1);
        const result = context.fn(args);
        delete context.fn;
        return context;
    }


    // apply
    Function.prototype.apply = function(context, args) {
        if (typeof this !== 'function') {
            throw new TypeError();
        }

        context = context || window;
        context.fn = this;
        const result = context.fn(args);
        delete context.fn;
        return result

    }

    // bind
    Function.prototype.myBind = function(context) {
        if (typeof this !== 'function') {
            throw new TypeError('error');
        }

        const args = [...arguments].slice(1);
        const fn = this;

        return function Fn() {
            return fn.apply(
                this instanceof Fn ? this : context,
                args.concat(...arguments)
            )
        }
    }
```
